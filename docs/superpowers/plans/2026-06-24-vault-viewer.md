**Goal:** Build an Obsidian plugin with a sidebar panel for browsing all file types, filtering/sorting, and rendering Office documents inline.

**Architecture:** Single plugin with a left sidebar `ItemView` (tree + file list) and a custom `View` for Office rendering. Services are decoupled from UI for testability.

**Tech Stack:** TypeScript, Obsidian API, esbuild, mammoth.js, SheetJS (xlsx), pptxjs

```typescript
// src/utils/extensions.ts
const OFFICE_EXTENSIONS = new Set([".docx", ".xlsx", ".pptx"]);
const IMAGE_EXTENSIONS = new Set([
  ".png", ".jpg", ".jpeg", ".gif", ".svg", ".webp", ".bmp",
]);

export function getExtension(filename: string): string {
  if (filename.endsWith(".excalidraw.md")) return ".excalidraw.md";
  if (filename.endsWith(".canvas.md")) return ".canvas.md";
  const dotIndex = filename.lastIndexOf(".");
  if (dotIndex === -1) return "";
  return filename.slice(dotIndex);
}

export function isTreeExtension(
  filename: string,
  treeExtensions: string[]
): boolean {
  return treeExtensions.some((ext) => filename.endsWith(ext));
}

export function isOfficeExtension(filename: string): boolean {
  const ext = getExtension(filename);
  return OFFICE_EXTENSIONS.has(ext);
}

export function isImageExtension(filename: string): boolean {
  const ext = getExtension(filename);
  return IMAGE_EXTENSIONS.has(ext);
}

export function isFileTypeVisible(
  extension: string,
  hiddenExtensions: string[]
): boolean {
  return !hiddenExtensions.includes(extension);
}

export type FileTypeCategory = "markdown" | "office" | "image" | "pdf" | "other";

export function getFileType(filename: string): FileTypeCategory {
  const ext = getExtension(filename);
  if (ext === ".md" || ext.endsWith(".md")) return "markdown";
  if (OFFICE_EXTENSIONS.has(ext)) return "office";
  if (IMAGE_EXTENSIONS.has(ext)) return "image";
  if (ext === ".pdf") return "pdf";
  return "other";
}
```

```typescript
// src/main.ts
import { Plugin } from "obsidian";
import { VaultViewerSettings, DEFAULT_SETTINGS, VaultViewerSettingTab } from "./settings";
import { VaultViewerView, VIEW_TYPE_VAULT_VIEWER } from "./views/VaultViewerView";

export default class VaultViewerPlugin extends Plugin {
  settings: VaultViewerSettings;

  async onload() {
    await this.loadSettings();

    this.addSettingTab(new VaultViewerSettingTab(this.app, this));

    this.registerView(
      VIEW_TYPE_VAULT_VIEWER,
      (leaf) => new VaultViewerView(leaf, this)
    );

    this.addCommand({
      id: "open-vault-viewer",
      name: "Open Vault Viewer",
      callback: () => this.activateView(),
    });

    this.app.workspace.onLayoutReady(() => {
      this.activateView();
    });
  }

  async loadSettings() {
    this.settings = Object.assign(
      {},
      DEFAULT_SETTINGS,
      await this.loadData()
    );
  }

  async saveSettings() {
    await this.saveData(this.settings);
  }

  async activateView() {
    const { workspace } = this.app;
    let leaf = workspace.getLeavesOfType(VIEW_TYPE_VAULT_VIEWER)[0];

    if (!leaf) {
      leaf = workspace.getLeftLeaf(false);
      if (!leaf) return;
      await leaf.setViewState({ type: VIEW_TYPE_VAULT_VIEWER, active: true });
    }

    workspace.revealLeaf(leaf);
  }

  onunload() {
    this.app.workspace.detachLeavesOfType(VIEW_TYPE_VAULT_VIEWER);
  }
}
```

```typescript
// src/services/FileService.ts
import { TFile, TFolder, Vault, TAbstractFile } from "obsidian";
import { isTreeExtension, getExtension } from "../utils/extensions";

export type SortField = "name" | "mtime" | "ctime" | "size";
export type SortOrder = "asc" | "desc";

export interface FileChangeCallback {
  (type: "create" | "delete" | "rename", file: TAbstractFile): void;
}

export class FileService {
  private vault: Vault;
  private listeners: (() => void)[] = [];
  private onChangeCallbacks: FileChangeCallback[] = [];

  constructor(vault: Vault) {
    this.vault = vault;
  }

  startListening(onLayoutReady: boolean, cb: FileChangeCallback): void {
    this.onChangeCallbacks.push(cb);

    if (onLayoutReady) {
      const ref = this.vault.on("create", (file) => cb("create", file));
      this.listeners.push(ref);
    } else {
      const ref = this.vault.on("delete", (file) => cb("delete", file));
      this.listeners.push(ref);
      this.listeners.push(
        this.vault.on("rename", (file, _oldPath) => cb("rename", file))
      );
    }
  }

  stopListening(): void {
    for (const ref of this.listeners) {
      ref();
    }
    this.listeners = [];
  }

  getAllFiles(): TFile[] {
    return this.vault.getFiles();
  }

  getTreeFiles(files: TFile[], treeExtensions: string[]): TFile[] {
    return files.filter((f) => isTreeExtension(f.name, treeExtensions));
  }

  getNonMdFiles(
    files: TFile[],
    treeExtensions: string[],
    sortField: SortField,
    sortOrder: SortOrder
  ): TFile[] {
    const filtered = files.filter(
      (f) => !isTreeExtension(f.name, treeExtensions)
    );
    return this.sortFiles(filtered, sortField, sortOrder);
  }

  getDirectoryChildren(dirPath: string, files: TFile[]): TFile[] {
    const prefix = dirPath.endsWith("/") ? dirPath : dirPath + "/";
    return files.filter((f) => {
      if (!f.path.startsWith(prefix)) return false;
      const rest = f.path.slice(prefix.length);
      return !rest.includes("/");
    });
  }

  sortFiles(
    files: TFile[],
    field: SortField,
    order: SortOrder
  ): TFile[] {
    const sorted = [...files].sort((a, b) => {
      let cmp = 0;
      switch (field) {
        case "name":
          cmp = a.name.localeCompare(b.name);
          break;
        case "mtime":
          cmp = a.stat.mtime - b.stat.mtime;
          break;
        case "ctime":
          cmp = a.stat.ctime - b.stat.ctime;
          break;
        case "size":
          cmp = a.stat.size - b.stat.size;
          break;
      }
      return order === "asc" ? cmp : -cmp;
    });
    return sorted;
  }

  getDirectSubfolders(
    rootPath: string,
    folders: TFolder[]
  ): TFolder[] {
    const prefix = rootPath.endsWith("/") ? rootPath : rootPath + "/";
    return folders.filter((f) => {
      if (f.path === rootPath) return false;
      if (!f.path.startsWith(prefix)) return false;
      const rest = f.path.slice(prefix.length);
      return !rest.includes("/");
    });
  }
}
```

```typescript
// src/services/LinkService.ts
import { TFile, MetadataCache } from "obsidian";

export interface ResolvedLink {
  file: TFile | null;
  linkType: "link" | "embed";
  displayText?: string;
  original: string;
  resolved: boolean;
}

export class LinkService {
  private cache: MetadataCache;
  private listeners: (() => void)[] = [];

  constructor(cache: MetadataCache) {
    this.cache = cache;
  }

  startListening(cb: (file: TFile) => void): void {
    this.listeners.push(
      this.cache.on("changed", (file: TFile) => cb(file))
    );
  }

  stopListening(): void {
    for (const ref of this.listeners) ref();
    this.listeners = [];
  }

  getForwardLinks(file: TFile): ResolvedLink[] {
    const cache = this.cache.getFileCache(file);
    if (!cache) return [];

    const results: ResolvedLink[] = [];

    if (cache.links) {
      for (const link of cache.links) {
        const dest = this.cache.getFirstLinkpathDest(link.link, file.path);
        results.push({
          file: dest,
          linkType: "link",
          displayText: link.displayText || undefined,
          original: link.original,
          resolved: dest !== null,
        });
      }
    }

    if (cache.embeds) {
      for (const embed of cache.embeds) {
        const dest = this.cache.getFirstLinkpathDest(embed.link, file.path);
        results.push({
          file: dest,
          linkType: "embed",
          original: embed.original,
          resolved: dest !== null,
        });
      }
    }

    return results;
  }

  isMarkdownFile(file: TFile): boolean {
    return file.extension === "md";
  }
}
```

```typescript
// src/services/OfficeRenderer.ts
import { TFile, Vault } from "obsidian";
import mammoth from "mammoth";
import * as XLSX from "xlsx";

export type OfficeType = "docx" | "xlsx" | "pptx";

export interface RenderResult {
  html: string;
  title: string;
}

export class OfficeRenderer {
  private vault: Vault;

  constructor(vault: Vault) {
    this.vault = vault;
  }

  async render(file: TFile): Promise<RenderResult> {
    const ext = file.extension.toLowerCase();
    const arrayBuffer = await this.vault.readBinary(file);

    switch (ext) {
      case "docx":
        return this.renderDocx(arrayBuffer, file.name);
      case "xlsx":
        return this.renderXlsx(arrayBuffer, file.name);
      case "pptx":
        return this.renderPptx(arrayBuffer, file.name);
      default:
        throw new Error(`Unsupported file type: .${ext}`);
    }
  }

  private async renderDocx(
    buffer: ArrayBuffer,
    filename: string
  ): Promise<RenderResult> {
    const result = await mammoth.convertToHtml({ arrayBuffer: buffer });
    return {
      html: `<div class="office-docx">${result.value}</div>`,
      title: filename,
    };
  }

  private async renderXlsx(
    buffer: ArrayBuffer,
    filename: string
  ): Promise<RenderResult> {
    const workbook = XLSX.read(buffer, { type: "array" });
    let html = '<div class="office-xlsx">';

    for (const sheetName of workbook.SheetNames) {
      const sheet = workbook.Sheets[sheetName];
      const json = XLSX.utils.sheet_to_json(sheet, { header: 1 }) as any[][];
      if (json.length === 0) continue;

      html += `<h3>${this.escapeHtml(sheetName)}</h3>`;
      html += '<table class="office-table">';

      for (const row of json) {
        html += "<tr>";
        for (const cell of row) {
          html += `<td>${this.escapeHtml(String(cell ?? ""))}</td>`;
        }
        html += "</tr>";
      }
      html += "</table>";
    }

    html += "</div>";
    return { html, title: filename };
  }

  private async renderPptx(
    buffer: ArrayBuffer,
    filename: string
  ): Promise<RenderResult> {
    const pptxgen = require("pptxjs");
    const result = await pptxgen.convertToHtml(buffer);
    return {
      html: `<div class="office-pptx">${result}</div>`,
      title: filename,
    };
  }

  private escapeHtml(text: string): string {
    const map: Record<string, string> = {
      "&": "&amp;",
      "<": "&lt;",
      ">": "&gt;",
      '"': "&quot;",
      "'": "&#039;",
    };
    return text.replace(/[&<>"']/g, (c) => map[c]);
  }
}
```

```javascript
const fs = require('fs');
const path = require('path');

const pluginDir = __dirname;
let fileWatcher;

function reloadPlugin() {
  const plugin = app.plugins.plugins['vault-viewer'];
  if (plugin) {
    plugin.unload();
    plugin.load();
  }
}

function startWatching() {
  if (fileWatcher) return;
  fileWatcher = fs.watch(path.join(pluginDir, 'main.js'), () => {
    setTimeout(reloadPlugin, 200);
  });
}

module.exports = () => ({
  id: 'vault-viewer-dev',
  name: 'Vault Viewer Dev',
  onload: () => startWatching(),
  onunload: () => { if (fileWatcher) fileWatcher.close(); },
});
```
